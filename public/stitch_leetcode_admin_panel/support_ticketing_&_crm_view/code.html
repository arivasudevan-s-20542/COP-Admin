def countSubarrays(nums, k):
    count = 0
    curr_sum = 0
    prefix_sums = {0: 1}
    
    for num in nums:
        curr_sum += num
        # Potential issue here with large inputs?
        if (curr_sum - k) in prefix_sums:
            count += prefix_sums[curr_sum - k]
            
    return count
